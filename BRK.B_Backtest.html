# Create the BT strategy. BRK.B owned from 2012-2021
fixed_weights = bt.Strategy('BRK.B',
                      [bt.algos.RunYearly(),
                       bt.algos.SelectAll(),
                       bt.algos.WeighSpecified(brkb=1), 
                       bt.algos.Rebalance()
                       ])
In [ ]:
# Create a backtest and run it
test = bt.Backtest(fixed_weights, data)
res = bt.run(test)
